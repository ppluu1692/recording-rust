// src/wav_writer.rs
use anyhow::Result;
use hound::{WavWriter, WavSpec, SampleFormat};
use std::sync::{Arc, Mutex};

pub type SharedWavWriter = Arc<Mutex<Option<WavWriter<std::io::BufWriter<std::fs::File>>>>>;

/// Tạo WAV writer với thông số cho trước
pub fn create_wav_writer(path: &str, channels: u16, sample_rate: u32) -> Result<WavWriter<std::io::BufWriter<std::fs::File>>> {
    let spec = WavSpec {
        channels,
        sample_rate,
        bits_per_sample: 32,
        sample_format: SampleFormat::Float,
    };
    let writer = WavWriter::create(path, spec)?;
    Ok(writer)
}

/// Bọc writer trong Arc<Mutex> để chia sẻ giữa các thread
pub fn wrap_writer(writer: WavWriter<std::io::BufWriter<std::fs::File>>) -> SharedWavWriter {
    Arc::new(Mutex::new(Some(writer)))
}

/// Ghi một batch samples vào writer
pub fn write_samples(shared: &SharedWavWriter, samples: &[f32]) -> Result<()> {
    if let Ok(mut guard) = shared.lock() {
        if let Some(writer) = guard.as_mut() {
            for &sample in samples {
                writer.write_sample(sample)?;
            }
        }
    }
    Ok(())
}

/// Finalize và đóng file WAV
pub fn finalize_writer(shared: &SharedWavWriter) -> Result<()> {
    if let Ok(mut guard) = shared.lock() {
        if let Some(writer) = guard.take() {
            writer.finalize()?;
        }
    }
    Ok(())
}

// src/recorder.rs
use anyhow::Result;
use cpal::{Device, SupportedStreamConfig, Stream, SampleFormat};
use cpal::traits::{DeviceTrait, StreamTrait};
use crate::wav_writer::{SharedWavWriter, write_samples};

/// Build input stream từ device và ghi vào WAV writer
pub fn build_input_stream(
    device: &Device,
    config: &SupportedStreamConfig,
    writer: SharedWavWriter,
    label: &str,  // "mic" hoặc "system" — để log
) -> Result<Stream> {
    let label = label.to_string();
    let err_label = label.clone();
    
    let stream = match config.sample_format() {
        SampleFormat::F32 => build_stream_f32(device, config, writer, label)?,
        SampleFormat::I16 => build_stream_i16(device, config, writer, label)?,
        SampleFormat::U16 => build_stream_u16(device, config, writer, label)?,
        _ => anyhow::bail!("Sample format không được hỗ trợ"),
    };
    
    Ok(stream)
}

fn build_stream_f32(device: &Device, config: &SupportedStreamConfig, writer: SharedWavWriter, label: String) -> Result<Stream> {
    let stream = device.build_input_stream(
        &config.config(),
        move |data: &[f32], _| {
            if let Err(e) = write_samples(&writer, data) {
                eprintln!("[{}] Lỗi ghi samples: {}", label, e);
            }
        },
        move |err| eprintln!("[{}] Stream error: {}", label, err),  // sửa err_label
        None,
    )?;
    Ok(stream)
}

fn build_stream_i16(device: &Device, config: &SupportedStreamConfig, writer: SharedWavWriter, label: String) -> Result<Stream> {
    let stream = device.build_input_stream(
        &config.config(),
        move |data: &[i16], _| {
            // Convert i16 -> f32 normalized [-1.0, 1.0]
            let converted: Vec<f32> = data.iter().map(|&s| s as f32 / i16::MAX as f32).collect();
            if let Err(e) = write_samples(&writer, &converted) {
                eprintln!("[{}] Lỗi ghi samples: {}", label, e);
            }
        },
        move |err| eprintln!("[{}] Stream error: {}", label, err),
        None,
    )?;
    Ok(stream)
}

fn build_stream_u16(device: &Device, config: &SupportedStreamConfig, writer: SharedWavWriter, label: String) -> Result<Stream> {
    let stream = device.build_input_stream(
        &config.config(),
        move |data: &[u16], _| {
            // Convert u16 -> f32 normalized [-1.0, 1.0]
            let converted: Vec<f32> = data.iter().map(|&s| (s as f32 / u16::MAX as f32) * 2.0 - 1.0).collect();
            if let Err(e) = write_samples(&writer, &converted) {
                eprintln!("[{}] Lỗi ghi samples: {}", label, e);
            }
        },
        move |err| eprintln!("[{}] Stream error: {}", label, err),
        None,
    )?;
    Ok(stream)
}

/// Bắt đầu stream
pub fn start_stream(stream: &Stream) -> Result<()> {
    stream.play()?;
    Ok(())
}

/// Dừng stream
pub fn stop_stream(stream: &Stream) -> Result<()> {
    stream.pause()?;
    Ok(())
}

// src/main.rs
mod audio_device;
mod recorder;
mod wav_writer;

use anyhow::Result;
use std::sync::{Arc, atomic::{AtomicBool, Ordering}};
use cpal::traits::StreamTrait;

fn main() -> Result<()> {
    println!("🎙️  Audio Recorder — Nhấn Ctrl+C để dừng\n");

    // 1. Khởi tạo host và liệt kê devices
    let host = audio_device::get_host();
    audio_device::list_input_devices(&host)?;
    audio_device::list_output_devices(&host)?;
    println!();

    // 2. Lấy devices
    let mic_device = audio_device::get_microphone(&host)?;
    let sys_device = audio_device::get_system_audio(&host)?;
    println!("🎤 Mic: {}", mic_device.name()?);
    println!("🔊 System: {}", sys_device.name()?);

    // 3. Lấy configs
    let mic_config = audio_device::get_input_config(&mic_device)?;
    let sys_config = audio_device::get_input_config(&sys_device)?;

    // 4. Tạo WAV writers
    let mic_writer = wav_writer::wrap_writer(wav_writer::create_wav_writer(
        "mic_output.wav",
        mic_config.channels(),
        mic_config.sample_rate().0,
    )?);

    let sys_writer = wav_writer::wrap_writer(wav_writer::create_wav_writer(
        "system_output.wav",
        sys_config.channels(),
        sys_config.sample_rate().0,
    )?);

    // 5. Build streams
    let mic_stream = recorder::build_input_stream(&mic_device, &mic_config, mic_writer.clone(), "mic")?;
    let sys_stream = recorder::build_input_stream(&sys_device, &sys_config, sys_writer.clone(), "system")?;

    // 6. Bắt đầu ghi
    mic_stream.play()?;
    sys_stream.play()?;
    println!("\n⏺️  Đang ghi âm...");

    // 7. Chờ Ctrl+C
    let running = Arc::new(AtomicBool::new(true));
    let r = running.clone();
    ctrlc::set_handler(move || {
        r.store(false, Ordering::SeqCst);
    })?;

    while running.load(Ordering::SeqCst) {
        std::thread::sleep(std::time::Duration::from_millis(100));
    }

    // 8. Dừng và lưu file
    println!("\n⏹️  Đang dừng...");
    mic_stream.pause()?;
    sys_stream.pause()?;

    wav_writer::finalize_writer(&mic_writer)?;
    wav_writer::finalize_writer(&sys_writer)?;

    println!("✅ Đã lưu: mic_output.wav và system_output.wav");
    Ok(())
}

Phân tích cách giao tiếp thread hiện tại
Trong code trên, tôi đang dùng Arc<Mutex<T>> — cụ thể là:

rust
pub type SharedWavWriter = Arc<Mutex<Option<WavWriter<...>>>>;
Luồng hoạt động hiện tại
Audio callback thread ──[lock Mutex]──► WavWriter ──► file.wav
Audio callback thread ──[lock Mutex]──► WavWriter ──► file.wav
Vấn đề: Audio callback được gọi với độ trễ cực thấp (real-time). Nếu Mutex đang bị lock bởi thread khác → callback bị block → audio bị drop/glitch.

So sánh các phương pháp
1. Arc<Mutex<T>> — đang dùng
rust
// Callback ghi thẳng vào file qua Mutex
move |data: &[f32], _| {
    let mut guard = writer.lock().unwrap(); // ❌ có thể block!
    guard.write_sample(sample)?;
}
✅	Đơn giản, dễ hiểu
❌	Mutex có thể block audio callback
❌	Ghi file I/O trong real-time thread — rất tệ
❌	lock() không có thời gian chờ cố định
2. crossbeam-channel — tốt hơn đáng kể
Audio callback ──[send]──► channel ──[recv]──► Writer thread ──► file.wav
   (real-time)    non-blocking        blocking ok here
rust
// Callback chỉ send, không bao giờ block
move |data: &[f32], _| {
    let _ = tx.send(data.to_vec()); // ✅ gần như không tốn thời gian
}

// Thread riêng lo việc ghi file
while let Ok(samples) = rx.recv() {
    writer.write_samples(&samples)?; // I/O ở đây là ok
}
✅	Callback không bao giờ block
✅	Tách biệt real-time thread và I/O thread
✅	crossbeam nhanh hơn std::sync::mpsc
⚠️	Nếu channel đầy → vẫn có thể drop samples
3. Lock-free ring buffer — tốt nhất cho audio
Đây là cách các ứng dụng audio chuyên nghiệp dùng (JUCE, PortAudio...):

Audio callback ──[write]──► RingBuffer ──[read]──► Writer thread
                lock-free!
Dùng crate ringbuf:

Dự án học tập, đơn giản    →  Arc<Mutex>  (hiện tại)

Phân tích chi tiết
Tại sao không dùng ringbuf luôn từ đầu?
Vì ringbuf có một điểm yếu quan trọng: khi bạn cần gửi nhiều loại lệnh khác nhau (không chỉ samples), nó bất tiện hơn crossbeam. Ví dụ sau này bạn muốn gửi cả StartRecord, StopRecord, FlushBuffer — với channel bạn làm ngay bằng enum, với ringbuf phải tự xây thêm lớp protocol.

ringbuf = "0.4"

let ring = HeapRb::<f32>::new(4096 * 8);
let (mut prod, mut cons) = ring.split();

// audio callback — wait-free
move |data: &[f32], _| {
    let pushed = prod.push_slice(data);
    if pushed < data.len() {
        // buffer đầy → drop samples, nhưng KHÔNG block
    }
}

// writer thread
loop {
    let n = cons.pop_slice(&mut buf);
    writer.write(&buf[..n])?;
}
crossbeam-channel — linh hoạt hơn, dễ mở rộng:

rust
// Cargo.toml
crossbeam-channel = "0.5"

// Có thể gửi cả lệnh điều khiển lẫn data
enum AudioMsg {
    Samples(Vec<f32>),
    Stop,
    Flush,
}

let (tx, rx) = crossbeam_channel::bounded(512);

// callback — gần như lock-free với bounded channel
move |data: &[f32], _| {
    let _ = tx.try_send(AudioMsg::Samples(data.to_vec()));
    // try_send không block, trả Err nếu đầy
}

// writer thread
while let Ok(msg) = rx.recv() {
    match msg {
        AudioMsg::Samples(s) => writer.write(&s)?,
        AudioMsg::Stop => break,
        AudioMsg::Flush => writer.flush()?,
    }
}

    Device, SampleFormat, Stream, SupportedStreamConfig,

/// Dung lượng ring buffer tính bằng số f32 samples.
/// 4096 * 16 ≈ 1.5 giây buffer ở 44100Hz stereo — đủ dư cho bất kỳ writer nào.
const RING_CAPACITY: usize = 4096 * 16;
 
/// Khoảng thời gian drain thread ngủ giữa các lần poll ring buffer.
/// 5ms là cân bằng tốt: latency thấp nhưng không waste CPU.
const DRAIN_INTERVAL_MS: u64 = 5;
 
/// Số samples đọc mỗi lần drain (một "batch").
const DRAIN_BATCH: usize = 1024;
 
/// Khởi tạo toàn bộ capture pipeline cho một device:
///   - Tạo ring buffer
///   - Build cpal input stream (callback ghi vào ring buffer)
///   - Spawn drain thread (đọc ring buffer, gửi qua channel)
///
/// Trả về Stream (phải giữ sống) và JoinHandle của drain thread.
pub fn start_capture(
    device: &Device,
    config: &SupportedStreamConfig,
    tx: Sender<AudioMessage>,
    label: &str,
) -> Result<(Stream, thread::JoinHandle<()>)> {
    let ring = HeapRb::<f32>::new(RING_CAPACITY);
    let (prod, cons) = ring.split();
 
    let stream = build_input_stream(device, config, prod, label)?;
    stream.play()?;
 
    let handle = spawn_drain_thread(cons, tx, label);
 
    Ok((stream, handle))
}
 
/// Dừng capture: pause stream rồi gửi Stop qua channel.
pub fn stop_capture(stream: &Stream, tx: &Sender<AudioMessage>) -> Result<()> {
    stream.pause()?;
    let _ = tx.send(AudioMessage::Stop);
    Ok(())
}
 
// ─── private ────────────────────────────────────────────────────────────────
 
type RingProd = ringbuf::wrap::caching::Caching<
    std::sync::Arc<ringbuf::storage::Heap<f32>>,
    true,
    ringbuf::wrap::direct::Direct<std::sync::Arc<ringbuf::storage::Heap<f32>>, true>,
>;
 
type RingCons = ringbuf::wrap::caching::Caching<
    std::sync::Arc<ringbuf::storage::Heap<f32>>,
    false,
    ringbuf::wrap::direct::Direct<std::sync::Arc<ringbuf::storage::Heap<f32>>, false>,
>;
 
/// Build cpal input stream với sample format tự động.
/// Callback chỉ làm một việc: push_slice vào ring buffer — wait-free.
fn build_input_stream(
    device: &Device,
    config: &SupportedStreamConfig,
    mut prod: RingProd,
    label: &str,
) -> Result<Stream> {
    let label_err = label.to_string();
 
    let stream = match config.sample_format() {
        SampleFormat::F32 => {
            device.build_input_stream(
                &config.config(),
                move |data: &[f32], _| {
                    let written = prod.push_slice(data);
                    if written < data.len() {
                        // Ring buffer đầy: drop phần dư, KHÔNG block.
                        // Sau này có thể emit metric / increment counter ở đây.
                    }
                },
                move |e| eprintln!("[{}] stream error: {}", label_err, e),
                None,
            )?
        }
 
        SampleFormat::I16 => {
            device.build_input_stream(
                &config.config(),
                move |data: &[i16], _| {
                    // Convert i16 → f32 normalized [-1.0, 1.0]
                    // to_vec để tránh alloc trong hot path; dùng SmallVec nếu cần tối ưu hơn.
                    let converted: Vec<f32> =
                        data.iter().map(|&s| s as f32 / i16::MAX as f32).collect();
                    let _ = prod.push_slice(&converted);
                },
                move |e| eprintln!("[{}] stream error: {}", label_err, e),
                None,
            )?
        }
 
        SampleFormat::U16 => {
            device.build_input_stream(
                &config.config(),
                move |data: &[u16], _| {
                    let converted: Vec<f32> = data
                        .iter()
                        .map(|&s| (s as f32 / u16::MAX as f32) * 2.0 - 1.0)
                        .collect();
                    let _ = prod.push_slice(&converted);
                },
                move |e| eprintln!("[{}] stream error: {}", label_err, e),
                None,
            )?
        }
 
        fmt => anyhow::bail!("Sample format {:?} không được hỗ trợ", fmt),
    };
 
    Ok(stream)
}
 
/// Spawn drain thread: poll ring buffer mỗi DRAIN_INTERVAL_MS,
/// gửi batches qua crossbeam-channel tới writer/stream thread.
///
/// Tại sao poll thay vì wait-free notify?
/// Đơn giản hơn và DRAIN_INTERVAL_MS = 5ms là đủ thấp.
/// Nếu sau này cần latency < 1ms, thay bằng Parker/Unparker của crossbeam.
fn spawn_drain_thread(
    mut cons: RingCons,
    tx: Sender<AudioMessage>,
    label: &str,
) -> thread::JoinHandle<()> {
    let label = label.to_string();
    let mut batch = vec![0f32; DRAIN_BATCH];
 
    thread::Builder::new()
        .name(format!("drain-{}", label))
        .spawn(move || loop {
            let n = cons.pop_slice(&mut batch);
 
            if n > 0 {
                let samples = batch[..n].to_vec();
                // Nếu channel đầy (writer quá chậm), try_send drop batch này.
                // Tốt hơn block drain thread — block sẽ backup vào ring buffer.
                if tx.try_send(AudioMessage::Samples(samples)).is_err() {
                    eprintln!("[{}] channel đầy, drop batch {} samples", label, n);
                }
            }
 
            // Kiểm tra channel còn sống không (writer đã drop sender kia).
            if tx.is_disconnected() {
                break;
            }
 
            thread::sleep(Duration::from_millis(DRAIN_INTERVAL_MS));
        })
        .expect("spawn drain thread thất bại")
}
